""" OHM-BLOX ENGINE v1.0.4 - NCEA SPECIAL EDITION Core Systems: Physics, Procedural Generation, Game State Management """ from ursina import * from ursina.prefabs.first_person_controller import FirstPersonController import random # ========================================== # 1. GLOBAL CONFIGURATION & CONSTANTS # ========================================== VERSION = "1.0.4" GRAVITY = 0.8 WALK_SPEED = 12 JUMP_HEIGHT = 2 WORLD_SIZE = 200 TOTAL_STAGES = 25 DEBUG_MODE = False # ========================================== # 2. CORE GAME ENGINE CLASS # ========================================== class OhmGame(Ursina): def __init__(self): super().__init__() window.title = f"Ohm-blox Engine {VERSION}" window.borderless = False window.fullscreen = False window.exit_button.visible = False window.fps_counter.enabled = True self.state = 'MENU' self.score = 0 self.high_voltage_detected = False # ========================================== # 3. ENVIRONMENT & LIGHTING # ========================================== class Environment: def __init__(self): self.sun = DirectionalLight() self.sun.look_at(Vec3(1, -1, 1)) self.sky = Sky(texture='sky_sunset') self.floor = Entity( model='plane', collider='box', scale=WORLD_SIZE, texture='grass', texture_scale=(20,20), color=color.dark_gray ) def update_sun(self): self.sun.rotation_x += time.dt * 1.5 # ========================================== # 4. INTERACTABLE OBJECTS (POWER BLOCKS) # ========================================== class PowerBlock(Entity): def __init__(self, pos=(0,0,0), v=230): super().__init__( model='cube', position=pos, scale=(2.5, 1, 2.5), collider='box', texture='white_cube', color=color.random_color() ) self.voltage = v self.base_y = pos[1] self.pulse_speed = random.uniform(1, 3) def update(self): # Hover effect using sine wave math self.y = self.base_y + math.sin(time.time() * self.pulse_speed) * 0.5 # ========================================== # 5. USER INTERFACE (HUD) # ========================================== class HUD: def __init__(self): self.title = Text( text='PROJECT: OHM-BLOX', position=(-0.85, 0.45), scale=2, color=color.yellow ) self.stats = Text( text='V: 112 | I: 2A | R: CALCULATING...', position=(-0.85, 0.38), color=color.cyan ) self.crosshair = Entity( model='quad', parent=camera.ui, scale=0.01, color=color.white ) def update_stats(self, r_val): self.stats.text = f'V: 112 | I: 2A | R: {r_val} Ohms' # ========================================== # 6. MAIN GAME LOOP & LOGIC # ========================================== app = OhmGame() env = Environment() ui = HUD() # Generate Procedural Course course_blocks = [] for i in range(TOTAL_STAGES): x = random.randint(-15, 15) y = i * 1.5 z = i * 8 block = PowerBlock(pos=(x, y, z)) course_blocks.append(block) # The Goal (PlayStation Logic) ps5_goal = Entity( model='cube', color=color.blue, scale=(5, 3, 5), position=(0, TOTAL_STAGES * 1.5, TOTAL_STAGES * 8 + 10), collider='box', texture='brick' ) ps5_label = Text(parent=ps5_goal, text='FINISH LINE', y=2, billboard=True, scale=12) # Initialize Player player = FirstPersonController( model='cube', z=-10, color=color.orange, speed=WALK_SPEED, jump_height=JUMP_HEIGHT ) def update(): env.update_sun() # Fall detection (Short Circuit) if player.y < -30: player.position = (0, 10, -10) ui.title.text = "SHORT CIRCUIT! TRY AGAIN" ui.title.color = color.red invoke(setattr, ui.title, 'text', 'PROJECT: OHM-BLOX', delay=2) invoke(setattr, ui.title, 'color', color.yellow, delay=2) # Win detection if distance(player.position, ps5_goal.position) < 6: ui.update_stats(56) ui.title.text = "LEVEL COMPLETE - ACHEIVED!" ui.title.color = color.green player.disable() # ========================================== # 7. EXECUTION # ========================================== if __name__ == '__main__': app.run()
